自动点名器——我的第一个python程序

我上初三的时候，给物理老师弄了一个自动点名器（这玩意儿前前后后用了我半年的时间-_-||，借口是主要是没有大段的时间去开发与debug），并且在单次程序运行期间不会重复点名，这个程序所用到有关Python的知识点很少，但是它却是我真正意义上的第一个Python程序。

以下为源码（为保证隐私，姓名已做匿名处理）

import tkinter as tk
import random
#创建窗口
window = tk.Tk()
window.title("自动点名器-By一名热心同学")#标题
window.geometry("1000x600")#窗口大小
window.lift()#窗口弹出时置顶
window.resizable(False,False)#禁止用户调整大小
var=tk.StringVar()#定义一个字符串变量

#载入名字
namelist=["张三",
"李四",
"王五",
"写bug的代码人"]
print(namelist)
namelen = len(namelist)
print(namelen)
y = random.randrange(0,namelen)
global aname
aname=namelist[y]
print(aname)
on_hit=False

#按钮的函数
def start():
    namelen = len(namelist)
    print(namelen)
    y = random.randrange(0,namelen)
    global aname
    aname=namelist[y]
    print(aname)
    del namelist[y]
    global on_hit   #声明全局变量
    if on_hit==False:
        on_hit=True
        var.set(aname)
    else:
        on_hit=False
        var.set(aname)


def about():
    windows = tk.Tk()
    windows.title("自动点名器-关于")  #标题
    windows.geometry("400x300")                 #窗口大小
    windows.lift()                              #窗口弹出时置顶
    windows.resizable(False,False)              #禁止用户调整大小

    m = """
本软件由写bug的代码人友情编写
本程序开发者开发配置如下
系统:windows7 SP1
环境:Python3.8.10 64bit
本程序已采取相关指施以防止恶意更改人名！
每次按点名按钮就会随机出现一个名字，且
重启软件前不会重复
联系作者:找本班物理课代表
欢迎大家访问我的网址:bugcoder.ml

本程序尚有大量不足之处，欢迎指出！
    """

    labout = tk.Label(windows,
             text = m,    #文本内容
             bg = "white",    #背景颜色
             font = ("楷体", 15),    #字体字号
             width = 90, height = 40   #标签长度
             )
    labout.pack()
    windows.mainloop()
#内容
#自定义一个Label标题
l = tk.Label(window,
             text = "猜猜谁是幸运儿？",    #文本内容
             bg = "white",    #背景颜色
             font = ("楷体", 35),    #字体字号
             width = 50, height = 2   #标签长度
             )
l.pack()
   
#名字显示框
l2 = tk.Label(window,
             textvariable = var,    #文本内容
             bg = "white",    #背景颜色
             font = ("楷体", 190),    #字体字号
             width = 8, height = 2   #标签长度
             )
l2.place(x = 0, y = 100)

#按钮
b1 = tk.Button(window,
               text = "关于",
               font = ("楷体", 20),    #字体字号
               width = 18, height = 4,
               command = about)
b1.place(x = 100, y = 90)

b2 = tk.Button(window,
               text = "开始",
               font = ("楷体", 20),    
               width = 18, height = 4,
               command = start)
b2.place(x = 650, y = 90)

window.mainloop()
